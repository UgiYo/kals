<!-- [KALS] -->
<script type="text/javascript" src="http://public-kals-2010.dlll.nccu.edu.tw/kals/web_apps/generic/loader/release"></script>
<script type="text/javascript">
KALS_CONFIG = {
    /**
     * 標註範圍指定區塊
     * @type {string,null} jQuery的指定語法(selector)
     *     請盡量以ID為指定區塊。
     *     例如指定ID名為「annotation_scope」的區塊，請輸入「#annotation_scope」
     *     如果是指定區有多個區塊，那麼只有第一個區塊能夠進行標註。
     *     預設是選擇全部的網頁。
     */
    annotation_scope_selector: "#scope"
};    
</script>
<!-- [/KALS] -->
